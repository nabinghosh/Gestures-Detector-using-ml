echo "read row"
read row
i=1

while [ $i -le $row ]
do
    j=1
    p=`expr $row - $i`
    while [ $j -le $p ]
    do
        j=`expr $j + 1`
        echo -n "  "
    done
    j=1
    k=`expr 2 \* $i - 1`
    while [ $j -le $k ]
    do
        j=`expr $j + 1`
        echo -n "* "
    done
    echo
   i=`expr $i + 1`
done
