echo "enter the size of the array :"
read n
i=0
echo "enter the elements:"
for ((i=0; i<n; i++))
do
    read t
    a[$i]=$t
done

max=${a[0]}
min=${a[0]}

for ((i=0; i<n; i++))
do
    if [ $max -lt ${a[i]} ]
    then
        max=${a[i]}
    fi
    if [ $min -gt ${a[i]} ]
    then
        min=${a[i]}
    fi
    #i=$((i+1))
done

echo "maximum:$max"
echo "minimum:$min"
