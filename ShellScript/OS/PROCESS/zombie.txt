#include <stdio.h>
#include <unistd.h>

int main() {
    pid_t p = fork();

    if (p < 0) 
    {
        printf("Error creating child process\n");
    } 
    else if (p== 0) 
    {
        printf("Child process started with PID: %d\n", getpid());
        sleep(5);

        printf("Child process terminated\n");
    } 
    else if (p > 0)
    {
        sleep(5);
        printf("Parent process ID: %d\n", getpid());
        printf("Parent process terminated\n");
    }

    return 0;
}
