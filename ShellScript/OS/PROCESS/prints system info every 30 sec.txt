#include <stdio.h>
#include <unistd.h>

int main() {
    int p = fork();

    if (p<0) 
    {
        printf("Error creating child process\n");
    } 
    else if (p == 0) {
        printf("Child process started with PID: %d\n", getpid());
        for(int i=0;i<5;i++)
            {
                system("ps -ax");
                sleep(30);
            }
        
        
    } 

    return 0;
}
