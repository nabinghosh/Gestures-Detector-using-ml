#!/bin/bash

echo "Enter a number: "
read num

# Calculate the sum of digits
sum=0
while [ $num -gt 0 ]
do
  rem=$((num % 10))
  sum=$((sum + rem))
  num=$((num / 10))
done

echo "Sum of digits: $sum"

# Reverse the number
rev=0
while [ $num -gt 0 ]
do
  rem=$((num % 10))
  rev=$((rev * 10 + rem))
  num=$((num / 10))
done

echo "Reverse of the number: $rev"
